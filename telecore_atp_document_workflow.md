## Database Schema for ATP Document Workflow

### Core Tables:
```sql
-- ATP Documents
CREATE TABLE atp_documents (
    document_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    site_id VARCHAR(100) REFERENCES sites(site_id),
    atp_type VARCHAR(20) NOT NULL, -- 'software' or 'hardware'
    filename VARCHAR(255),
    file_path TEXT,
    file_size BIGINT,
    upload_date TIMESTAMP DEFAULT NOW(),
    checksum VARCHAR(64),
    version INTEGER DEFAULT 1,
    is_corrected BOOLEAN DEFAULT FALSE,
    status VARCHAR(50) DEFAULT 'uploaded'
);

-- Checklist Evaluations  
CREATE TABLE checklist_evaluations (
    evaluation_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    document_id UUID REFERENCES atp_documents(document_id),
    checklist_item_id VARCHAR(50),
    result VARCHAR(10), -- 'pass', 'fail', 'na'
    severity VARCHAR(20), -- 'critical', 'major', 'minor'
    comments TEXT,
    evidence_images JSONB,
    evaluated_by VARCHAR(100),
    evaluated_at TIMESTAMP DEFAULT NOW()
);

-- Punchlist Items
CREATE TABLE punchlist_items (
    punchlist_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    document_id UUID REFERENCES atp_documents(document_id),
    item_description TEXT,
    severity VARCHAR(20),
    status VARCHAR(20) DEFAULT 'open', -- 'open', 'in_progress', 'completed'
    assigned_engineer VARCHAR(100),
    before_evidence JSONB,
    after_evidence JSONB,
    rectification_notes TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    completed_at TIMESTAMP
);

-- Approval Workflow
CREATE TABLE approval_history (
    approval_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    document_id UUID REFERENCES atp_documents(document_id),
    approval_level VARCHAR(50), -- 'back_office', 'sme', 'head_noc'
    approver_name VARCHAR(100),
    decision VARCHAR(20), -- 'approved', 'rejected', 'approved_with_punchlist'
    comments TEXT,
    approved_at TIMESTAMP DEFAULT NOW()
);

-- Signature Transfer Records
CREATE TABLE signature_transfers (
    transfer_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    from_document_id UUID REFERENCES atp_documents(document_id),
    to_document_id UUID REFERENCES atp_documents(document_id),
    signatures_transferred JSONB,
    transfer_date TIMESTAMP DEFAULT NOW(),
    transfer_reason TEXT
);
```

### API Endpoints:
```javascript
// Document Management
POST /api/v1/task-management/atp-tasks/{siteId}/upload
GET  /api/v1/task-management/atp-tasks/pending
POST /api/v1/task-management/atp-tasks/bulk-upload
PUT  /api/v1/task-management/atp-tasks/{taskId}/submit

// Approval Workflow
GET  /api/v1/task-management/atp-approvals/queue
GET  /api/v1/task-management/atp-approvals/review/{documentId}
POST /api/v1/task-management/atp-approvals/{documentId}/evaluate
POST /api/v1/task-management/atp-approvals/{documentId}/decision

// Punchlist Management
GET  /api/v1/task-management/punchlist/tasks
GET  /api/v1/task-management/punchlist/rectify/{taskId}
POST /api/v1/task-management/punchlist/{punchlistId}/complete
POST /api/v1/task-management/punchlist/{taskId}/submit-corrected

// Final Approval & History
GET  /api/v1/task-management/atp-approvals/final-approval/{documentId}
POST /api/v1/task-management/signature-transfer/{documentId}
GET  /api/v1/task-management/atp-tasks/history/{taskId}
```
